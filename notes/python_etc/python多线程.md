**multi-threading**

```python
import time
import threading

def calc_square(numbers):
    print("calculating square numbers")
    for n in numbers:
        time.sleep(0.2)
        print('square:', n**2)
    
def calc_cube(numbers):
    print("calculating cube of numbers")
    for n in numbers:
        time.sleep(0.2)
        print('cube:', n**3)
        
arr = [2,3,4,5]

t1 = theading.Thread(target= calc_square, args=(arr,))
t2 = theading.Thread(target= calc_cube, args=(arr,))
t1.start()
t2.start()

t1.join()
t2.join()

calc_sqare(arr)
calc_cube(arr)
```
